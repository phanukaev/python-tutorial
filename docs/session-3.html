import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

data = pd.read_excel('./Sodium Data - KAF.xlsx')

data = np.array(data)
ss_data = np.array(data[2:17, 2:6], dtype=float)
ncf_data = np.array(data[18:40, 2:6], dtype=float)
pbs_data = np.array(data[41:49, 2:6], dtype=float)

def nansterr(array, axis = None):
    sizes = np.sum(~np.isnan(array), axis=axis)
    return np.nanstd(array, axis=axis, ddof = 1) / np.sqrt(sizes)

def compute_stats(array):
    means = np.nanmean(array, axis=0)
    stdevs = np.nanstd(array, axis=0, ddof=1)
    sterrs = nansterr(array, axis=0)

    return {'mean': means, 'stdev': stdevs, 'sterr': sterrs}

def make_plot(array):
    data = compute_stats(array)
    width = 1
    labels = ['BL', 'HS7', 'HS14', 'HS21']
    plt.bar(labels, 'mean', color='teal', yerr='sterr', capsize=6, data=data)
    plt.show()

def multiplot():
    x_points = np.linspace(1,4,num=4)
    width = 0.2
    sep = 0.05
    capsize = 3
    ax = plt.subplot()
    ax.bar(x_points + 0 * (width+sep), 'mean', label='SS-C', color='xkcd:cornflower blue',
            width=width, yerr='sterr', capsize=capsize, data=compute_stats(ss_data),
            )
    ax.bar(x_points + 1 * (width+sep), 'mean', label='NCF', color='xkcd:rust',
            width=width, yerr='sterr', capsize=capsize, data=compute_stats(ncf_data),
            )
    ax.bar(x_points + 2 * (width+sep), 'mean', label='PBS', color='xkcd:dark sea green',
            width=width, yerr='sterr', capsize=capsize, data=compute_stats(pbs_data),
            )
    ax.set_title('Na Excretion (mEq/day)')
    ax.set_xticks(x_points + width + sep)
    ax.set_xticklabels(['BL', 'HS7', 'HS14', 'HS21'])
    ax.legend()
    plt.show()

print(compute_stats(ss_data))
print(compute_stats(ncf_data))
print(compute_stats(pbs_data))
multiplot()
